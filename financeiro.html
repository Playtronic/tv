// Cloud Function: transferToChannel const axios = require('axios'); const admin = require('firebase-admin'); exports.transferToChannel = functions.https.onRequest(async (req, res) => { try { const { channelUserId, amount, idempotencyKey, adminEmail } = req.body; // Buscar token do canal no Firestore const channelDoc = await admin.firestore() .collection('channels') .where('pixDonations.mpUserId', '==', channelUserId) .limit(1) .get(); if (channelDoc.empty) { return res.status(404).json({ success: false, error: 'Canal não encontrado' }); } const channelData = channelDoc.docs[0].data(); const channelToken = channelData.pixDonations.mpAccessToken; // Fazer transferência via Mercado Pago API const transferResponse = await axios({ method: 'post', url: 'https://api.mercadopago.com/v1/transfers', headers: { 'Authorization': `Bearer ${channelToken}`, 'Content-Type': 'application/json', 'X-Idempotency-Key': idempotencyKey }, data: { amount: parseFloat(amount), user_id: channelUserId.toString() } }); return res.status(200).json({ success: true, transferId: transferResponse.data.id, amount: amount, message: 'Transferência realizada com sucesso' }); } catch (error) { console.error('Erro na transferência:', error); return res.status(500).json({ success: false, error: error.message }); } });
