// Função para calcular slot atual no modo PLAYLIST (SIMPLIFICADA) function calcularSlotAtualPlaylist() { const agora = new Date(); // Data da estreia (às 6h) const [anoEstreia, mesEstreia, diaEstreia] = RELEASE_DATE.split('-').map(Number); const dataEstreia = new Date(anoEstreia, mesEstreia - 1, diaEstreia, 6, 0, 0); // Diferença em minutos entre a estreia e agora const diferencaMs = agora.getTime() - dataEstreia.getTime(); const minutosDesdeEstreia = Math.floor(diferencaMs / (1000 * 60)); // Se for negativo, ainda não começou if (minutosDesdeEstreia < 0) { return 1; } // Cada slot tem 25 minutos, o slot 1 começa em 0 minutos let slotAtual = Math.floor(minutosDesdeEstreia / SLOT_DURATION) + 1; return slotAtual; }
